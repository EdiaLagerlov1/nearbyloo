// Add this with your other event listeners const changeUserBtn = document.getElementById('change-user-btn'); changeUserBtn.addEventListener('click', function() { // Generate a new random user ID const newUserId = 'user_' + Math.random().toString(36).substring(2, 15); // Store the new ID localStorage.setItem('cc_user_id', newUserId); // Log the user ID change sendLog('user_id_change', 'user_action', { oldUserId: localStorage.getItem('cc_user_id'), newUserId: newUserId }); // Provide visual feedback alert('User ID changed to: ' + newUserId); });
